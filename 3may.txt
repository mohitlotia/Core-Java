int calculator(char a,float num1,float num2){
    
        float result;
         
        if (a == 'a') {
        result = num1 + num2;
        printf("Sum: %.2f\n", result);
        } else if (a == 's') {
            result = num1 - num2;
            printf("Difference: %.2f\n", result);
        } else if (a == 'm') {
            result = num1 * num2;
            printf("Product: %.2f\n", result);
        } else if (a == 'o') {
            result = (int)num1 % (int)num2; 
            printf("Modular: %d\n", (int)result);
        } else if (a == 'd') {
            if (num2 != 0) {
                 result = num1 / num2;
                printf("Division: %.2f\n", result);
            } else {
                printf("Error: Division by zero\n");
            }
         }
         return 0;
}

int amrstrong(int b){
    int rem, sum = 0;
        
        int temp = b;
        while(b>0) {
            rem = b % 10;
            sum=sum+rem*rem*rem;
            b=b/10;
        }
    
        if (temp==sum){
            printf("%d is Armstrong.", sum);
        } else {
            printf("Not Armstrong.");
        }
        return 0;
}

int factorial(int n){
     int fact=1;
    
    
    for(int i=n;i>=1;i--){
        fact = fact*i;        
    }
    return fact;
}

int swap(int num1,int num2){
  
    int temp;

    temp = num1;
    num1 = num2;
    num2 = temp;

    printf("After swapping: num1 = %d, num2 = %d\n", num1, num2);

    return 0;
}

int fibonacci(int n){
    int n1=0,n2=1,n3;
    n=n-2;
    printf("%d %d ",n1,n2);
    
    for(int i=1;i<=n;i++){
        n3 = n2+n1;
        printf("%d ",n3);
        n1=n2;
        n2=n3;
        
    }
    return 0;
}
