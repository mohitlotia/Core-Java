import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Enter a positive integer: ");
        int number = scanner.nextInt();

        if (number <= 0) {
            System.out.println("Please enter a positive integer.");
        } else {
            System.out.println("Multiplication Table for " + number + ":");
            for (int i = 1; i <= 10; i++) {
                System.out.println(number + " x " + i + " = " + (number * i));
            }
        }

    }

import java.util.Scanner;

public class Mohit {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.println("choose  armstrong(A),  pallindrome(P), reverse(R) ");
        char checkType = sc.nextLine().charAt(0);

        if(checkType=='A'){
            System.out.println("enter the number to check armstrong");
            int num=sc.nextInt();
            int temp,rem,sum=0;
            temp=num;
            while (num>0){
                rem=num%10;
                sum=sum+rem*rem*rem;
                num=num/10;
            }if(sum==temp){
                System.out.println("armstrong number");
            }
            else{
                System.out.println("not armstrong number");
            }

        }
        if(checkType=='P'){
            System.out.println("enter the number to check pallindrome");
            int num=sc.nextInt();
            int temp,rem,sum=0;
            temp=num;
            while(num>0){
                rem=num%10;
                sum=sum*10+rem;
                num=num/10;
            }if(sum==temp){
                System.out.println("pallindrome number");
            }
            else{
                System.out.println("not pallindrome number");
            }

        }
        if(checkType=='R'){
            System.out.println("enter the number to check reverse ");
            int num=sc.nextInt();
            int temp,rem,sum=0;
            temp=num;
            while (num>0){
                rem=num%10;
                sum=sum*10+rem;
                num=num/10;

            }if(sum==temp){
                System.out.println("reverse number");
            }
            else{
                System.out.println("not reverse number");
            }

        }
    }
}
